cmake_minimum_required(VERSION 3.20)

project(manzil)

set(CMAKE_CXX_STANDARD 23)

set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

add_executable(${PROJECT_NAME} src/main.cpp)

target_compile_options(
  ${PROJECT_NAME}
  PRIVATE
    -g
    -pedantic
    -Wall
    -Weffc++
    -Wextra
    -Wconversion
    -Wsign-conversion)
